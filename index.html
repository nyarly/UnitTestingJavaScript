Sinon.JS

var clock;

beforeEach(function () {
  clock = sinon.useFakeTimers();
});

afterEach(function () {
  clock.restore();
});

it("calls back after 100ms": function () {
  var callback = sinon.spy();
  setTimeout(100, callback);

  expect(callback.notCalled).to.beTrue();
  clock.tick(100);
  expect(callback.calledOnce).to.beTrue();
});
